Sidenav

A vertical side navigation panel.

Usage

  • Link item with icon
  • Link item without icon
  • Link item truncated text is too long
  • Custom item
<Sidenav>
  <SidenavHeader>
    <Logo color="white" />
  </SidenavHeader>
  <SidenavMenu>
    <SidenavLinkItem>
      <a href="#">
        <HelpIcon /> Link item with icon
      </a>
    </SidenavLinkItem>
    <SidenavLinkItem>
      <a href="#">Link item without icon</a>
    </SidenavLinkItem>
    <SidenavLinkItem>
      <a href="#">
        <HelpIcon /> Link item truncated text is too long
      </a>
    </SidenavLinkItem>
    <SidenavItem>
      <span className="custom">Custom item</span>
    </SidenavItem>
  </SidenavMenu>
</Sidenav>
